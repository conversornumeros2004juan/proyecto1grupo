int main() {
    int a, b, c;
    int n = 4;

    for (a = 1; a <= n - 1; a++) {
        for (b = 1; b <= n - 1; b++) {
            for (c = 1; c <= n - 1; c++) {
                if (a * a + b * b == c * c) {
                    std::cout << "terna pitagorica (" << a << "," << b << "," << c << ")" << endl;
                }
            }
        }
    }

    return 0;
}
